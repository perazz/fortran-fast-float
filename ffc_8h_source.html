502#if defined(__GNUC__) || defined(__clang__)

551uint64_t ffc_umul128_generic(uint64_t ab, uint64_t cd, uint64_t *hi) {
552 uint64_t ad = ffc_emulu((uint32_t)(ab >> 32), (uint32_t)cd);
553 uint64_t bd = ffc_emulu((uint32_t)ab, (uint32_t)cd);
554 uint64_t adbc = ad + ffc_emulu((uint32_t)ab, (uint32_t)(cd >> 32));
555 uint64_t adbc_carry = (uint64_t)(adbc < ad);
556 uint64_t lo = bd + (adbc << 32);
557 *hi = ffc_emulu((uint32_t)(ab >> 32), (uint32_t)(cd >> 32)) + (adbc >> 32) +
558 (adbc_carry << 32) + (uint64_t)(lo < bd);
559 return lo;

709bool ffc_strncasecmp3(char const *actual_mixedcase, char const *expected_lowercase, size_t char_width) {
710 uint64_t mask = 0;
711 if (char_width == 1) { mask = 0x2020202020202020; }
712 else if (char_width == 2) { mask = 0x0020002000200020; }
713 else if (char_width == 4) { mask = 0x0000002000000020; }
714 else { return false; }
715
716 uint64_t val1 = 0;
717 uint64_t val2 = 0;
718 if (char_width == 1 || char_width == 2) {
719 memcpy(&val1, actual_mixedcase, 3 * char_width);
720 memcpy(&val2, expected_lowercase, 3 * char_width);
721 val1 |= mask;
722 val2 |= mask;
723 return val1 == val2;
724 } else if (char_width == 4) {
725 memcpy(&val1, actual_mixedcase, 2 * char_width);
726 memcpy(&val2, expected_lowercase, 2 * char_width);
727 val1 |= mask;
728 if (val1 != val2) {
729 return false;
730 }
731 return (actual_mixedcase[2] | 32) == (expected_lowercase[2]);
732 } else {
733 return false;
734 }
735}

783ffc_internal ffc_inline
784bool ffc_rounds_to_nearest(void) {
785 // https://lemire.me/blog/2020/06/26/gcc-not-nearest/
786#if (FLT_EVAL_METHOD != 1) && (FLT_EVAL_METHOD != 0)
787 return false;
788#endif
789 // See
790 // A fast function to check your floating-point rounding mode
791 // https://lemire.me/blog/2022/11/16/a-fast-function-to-check-your-floating-point-rounding-mode/
792 //
793 // This function is meant to be equivalent to :
794 // prior: #include <cfenv>
795 // return fegetround() == FE_TONEAREST;
796 // However, it is expected to be much faster than the fegetround()
797 // function call.
798 //
799 // The volatile keyword prevents the compiler from computing the function
800 // at compile-time.
801 // There might be other ways to prevent compile-time optimizations (e.g.,
802 // asm). The value does not need to be std::numeric_limits<float>::min(), any
803 // small value so that 1 + x should round to 1 would do (after accounting for
804 // excess precision, as in 387 instructions).
805 static float volatile fmin = FLT_MIN;
806 float fmini = fmin; // we copy it so that it gets loaded at most once.
807//
808// Explanation:
809// Only when fegetround() == FE_TONEAREST do we have that
810// fmin + 1.0f == 1.0f - fmin.
811//
812// FE_UPWARD:
813// fmin + 1.0f > 1
814// 1.0f - fmin == 1
815//
816// FE_DOWNWARD or FE_TOWARDZERO:
817// fmin + 1.0f == 1
818// 1.0f - fmin < 1
819//
820// Note: This may fail to be accurate if fast-math has been
821// enabled, as rounding conventions may not apply.
822#ifdef FFC_VISUAL_STUDIO
823#pragma warning(push)
824// todo: is there a VS warning?
825// see
826// https://stackoverflow.com/questions/46079446/is-there-a-warning-for-floating-point-equality-checking-in-visual-studio-2013
827#elif defined(__clang__)
828#pragma clang diagnostic push
829#pragma clang diagnostic ignored "-Wfloat-equal"
830#elif defined(__GNUC__)
831#pragma GCC diagnostic push
832#pragma GCC diagnostic ignored "-Wfloat-equal"
833#endif
834 return (fmini + 1.0f == 1.0f - fmini);
835#ifdef FFC_VISUAL_STUDIO
836#pragma warning(pop)
837#elif defined(__clang__)
838#pragma clang diagnostic pop
839#elif defined(__GNUC__)
840#pragma GCC diagnostic pop
841#endif
842}

1016ffc_parse_eight_digits_unrolled_swar(uint64_t val) {
1017 uint64_t const mask = 0x000000FF000000FF;
1018 uint64_t const mul1 = 0x000F424000000064; // 100 + (1000000ULL << 32)

1022 val = (((val & mask) * mul1) + (((val >> 16) & mask) * mul2)) >> 32;
1023 return (uint32_t)val;

1078 // (x - '0') <= 9
1079 // http://0x80.pl/articles/simd-parsing-int-sequences.html
1080 uint16x8_t const t0 = vsubq_u16(data, vmovq_n_u16('0'));

1085 return true;
1086 } else
1087 return false;

1091 (void)i;
1092 return false;

1147ffc_internal ffc_inline
1148ffc_parsed ffc_parse_number_string(
1149 char const *p,
1150 // CONTRACT: p < pend
1151 char const *pend,
1152 ffc_parse_options const options,
1153 // explicitly passed to encourage optimizer to specialize
1154 bool const basic_json_fmt
1155) {
1156 ffc_format fmt = options.format;
1157 char decimal_point = options.decimal_point;
1158
1159 FFC_DEBUG_ASSERT(fmt != 0);
1160
1161 ffc_parsed answer = {0};
1162 answer.negative = (*p == '-');
1163 // C++17 20.19.3.(7.1) explicitly forbids '+' sign here
1164 // so we only allow it if we've been told to, and are not in json mode
1165 bool allow_leading_plus = fmt & FFC_FORMAT_FLAG_ALLOW_LEADING_PLUS;
1166 if ((*p == '-') || (uint64_t)(allow_leading_plus && !basic_json_fmt && *p == '+')) {
1167 ++p;
1168 if (p == pend) {
1169 return ffc_report_parse_error(p, FFC_PARSE_OUTCOME_MISSING_INTEGER_OR_DOT_AFTER_SIGN);
1170 }
1171 if (basic_json_fmt) {
1172 if (!ffc_is_integer(*p)) { // a sign must be followed by an integer
1173 return ffc_report_parse_error(p, FFC_PARSE_OUTCOME_JSON_MISSING_INTEGER_AFTER_SIGN);
1174 }
1175 } else {
1176 // a sign must be followed by an integer or the dot
1177 if (!ffc_is_integer(*p) && (*p != decimal_point)) {
1178 return ffc_report_parse_error(p, FFC_PARSE_OUTCOME_MISSING_INTEGER_OR_DOT_AFTER_SIGN);
1179 }
1180 }
1181 }
1182
1183 // phew, we've found the digits
1184 char const *const start_digits = p;
1185
1186 uint64_t i = 0; // an unsigned int avoids signed overflows (which are bad)
1187
1188 while ((p != pend) && ffc_is_integer(*p)) {
1189 // Horner's method: only ever multiplies by the constant 10
1190 // avoiding variable power-of-10 multiplies
1191
1192 // might overflow, we will handle the overflow later
1193 uint64_t digit_value = (uint64_t)(*p - '0');
1194 i = (10 * i) + digit_value;
1195 ++p;
1196 }
1197
1198 char const *const end_of_integer_part = p;
1199
1200 int64_t digit_count = (int64_t)(end_of_integer_part - start_digits);
1201 answer.int_part_start = (char*)start_digits;
1202 answer.int_part_len = (size_t)(digit_count);
1203
1204 if (basic_json_fmt) {
1205 // at least 1 digit in integer part
1206 if (digit_count == 0) {
1207 return ffc_report_parse_error(p, FFC_PARSE_OUTCOME_JSON_NO_DIGITS_IN_INTEGER_PART);
1208 }
1209 // no leading zeros
1210 if ((start_digits[0] == '0' && digit_count > 1)) {
1211 return ffc_report_parse_error(start_digits, FFC_PARSE_OUTCOME_JSON_LEADING_ZEROS_IN_INTEGER_PART);
1212 }
1213 }
1214
1215 int64_t exponent = 0;
1216 bool const has_decimal_point = (p != pend) && (*p == decimal_point);
1217
1218 /* post-decimal exponential part (calculates a negative exponent) */
1219 if (has_decimal_point) {
1220 ++p;
1221 char const *before = p;
1222 // can occur at most twice without overflowing, but let it occur more, since
1223 // for integers with many digits, digit parsing is the primary bottleneck.
1224 ffc_loop_parse_if_eight_digits(&p, pend, &i);
1225
1226 while ((p != pend) && ffc_is_integer(*p)) {
1227 uint8_t digit = (uint8_t)(*p - (char)('0'));
1228 ++p;
1229 i = i * 10 + digit; // in rare cases, this will overflow, but that's ok
1230 }
1231
1232 // pre: i = 123, digit_count = 3
1233 // 123.456
1234 // ^ pend
1235 // ^ p
1236 // ^ before
1237 // exponent = -3
1238 // i = 123456
1239 // digit_count = 3 - (-3) = 6
1240 exponent = before - p;
1241 answer.fraction_part_start = (char*)before;
1242 answer.fraction_part_len = (size_t)(p - before);
1243 digit_count -= exponent;
1244 }
1245 if (basic_json_fmt) {
1246 // at least 1 digit in fractional part
1247 if (has_decimal_point && exponent == 0) {
1248 return ffc_report_parse_error(p, FFC_PARSE_OUTCOME_JSON_NO_DIGITS_IN_FRACTIONAL_PART);
1249 }
1250 } else if (digit_count == 0) { // we must have encountered at least one integer!
1251 return ffc_report_parse_error(p, FFC_PARSE_OUTCOME_NO_DIGITS_IN_MANTISSA);
1252 }
1253
1254 /* explicit exponential part */
1255 int64_t exp_number = 0;
1256 if (((uint64_t)(fmt & FFC_FORMAT_FLAG_SCIENTIFIC) && (p != pend) &&
1257 (('e' == *p) || ('E' == *p))) ||
1258 ((uint64_t)(fmt & FFC_FORMAT_FLAG_BASIC_FORTRAN) && (p != pend) &&
1259 (('+' == *p) || ('-' == *p) || ('d' == *p) ||
1260 ('D' == *p)))) {
1261 char const *location_of_e = p;
1262 if (('e' == *p) || ('E' == *p) || ('d' == *p) ||
1263 ('D' == *p)) {
1264 ++p;
1265 }
1266 bool neg_exp = false;
1267 if ((p != pend) && ('-' == *p)) {
1268 neg_exp = true;
1269 ++p;
1270 } else if ((p != pend) && ('+' == *p)) { // '+' on exponent is allowed by C++17 20.19.3.(7.1)
1271 ++p;
1272 }
1273 if ((p == pend) || !ffc_is_integer(*p)) {
1274 if (!(uint64_t)(fmt & FFC_FORMAT_FLAG_FIXED)) {
1275 // The exponential part is invalid for scientific notation, so it must
1276 // be a trailing token for fixed notation. However, fixed notation is
1277 // disabled, so report a scientific notation error.
1278 return ffc_report_parse_error(p, FFC_PARSE_OUTCOME_MISSING_EXPONENTIAL_PART);
1279 }
1280 // Otherwise, we will be ignoring the 'e'.
1281 p = location_of_e;
1282 } else {
1283 while ((p != pend) && ffc_is_integer(*p)) {
1284 uint8_t digit = (uint8_t)(*p - '0');
1285 if (exp_number < 0x10000000) {
1286 exp_number = 10 * exp_number + digit;
1287 }
1288 ++p;
1289 }
1290 if (neg_exp) {
1291 exp_number = -exp_number;
1292 }
1293 exponent += exp_number;
1294 }
1295 } else {
1296 // If it scientific and not fixed, we have to bail out.
1297 if ((uint64_t)(fmt & FFC_FORMAT_FLAG_SCIENTIFIC) &&
1298 !(uint64_t)(fmt & FFC_FORMAT_FLAG_FIXED)) {
1299 return ffc_report_parse_error(p, FFC_PARSE_OUTCOME_MISSING_EXPONENTIAL_PART);
1300 }
1301 }
1302 answer.lastmatch = p;
1303 answer.valid = true;
1304
1305 // If we frequently had to deal with long strings of digits,
1306 // we could extend our code by using a 128-bit integer instead
1307 // of a 64-bit integer. However, this is uncommon.
1308 //
1309 // We can deal with up to 19 digits.
1310 if (digit_count > 19) { // this is uncommon
1311 ffc_debug("high digit_count %lld\n", digit_count);
1312 // It is possible that the integer had an overflow.
1313 // We have to handle the case where we have 0.0000somenumber.
1314 // We need to be mindful of the case where we only have zeroes...
1315 // E.g., 0.000000000...000.
1316 char const *start = start_digits;
1317 while ((start != pend) && (*start == '0' || *start == decimal_point)) {
1318 if (*start == '0') {
1319 digit_count--;
1320 }
1321 start++;
1322 }
1323
1324 if (digit_count > 19) {
1325 answer.too_many_digits = true;
1326 // Let us start again, this time, avoiding overflows.
1327 // We don't need to call if is_integer, since we use the
1328 // pre-tokenized spans from above.
1329 i = 0;
1330 p = answer.int_part_start;
1331 char const *int_end = p + answer.int_part_len;
1332 uint64_t const minimal_nineteen_digit_integer = 1000000000000000000;
1333 while ((i < minimal_nineteen_digit_integer) && (p != int_end)) {
1334 i = i * 10 + (uint64_t)(*p - '0');
1335 ++p;
1336 }
1337 if (i >= minimal_nineteen_digit_integer) { // We have a big integer
1338 exponent = end_of_integer_part - p + exp_number;
1339 } else { // We have a value with a fractional component.
1340 p = answer.fraction_part_start;
1341 char const *frac_end = p + answer.fraction_part_len;
1342 while ((i < minimal_nineteen_digit_integer) && (p != frac_end)) {
1343 i = i * 10 + (uint64_t)(*p - '0');
1344 ++p;
1345 }
1346 exponent = answer.fraction_part_start - p + exp_number;
1347 }
1348 // We have now corrected both exponent and i, to a truncated value
1349 }
1350 }
1351 answer.exponent = exponent;
1352 answer.mantissa = i;
1353 return answer;
1354}

1361ffc_internal ffc_inline
1362ffc_result ffc_parse_infnan(
1363 char *first, char *last,
1364 ffc_value *value,
1365 ffc_value_kind vk,
1366 ffc_format fmt
1367) {
1368 ffc_debug("parse_infnan\n");
1369 ffc_result answer;
1370 answer.ptr = first;
1371 answer.outcome = FFC_OUTCOME_OK; // be optimistic
1372 // assume first < last, so dereference without checks;
1373 bool const minus_sign = (*first == '-');
1374 // C++17 20.19.3.(7.1) explicitly forbids '+' sign here
1375 if ((*first == '-') ||
1376 ((uint64_t)(fmt & FFC_FORMAT_FLAG_ALLOW_LEADING_PLUS) &&
1377 (*first == '+'))) {
1378 ++first;
1379 }
1380 if (last - first >= 3) {
1381 if (ffc_strncasecmp3(first, "nan", 1)) {
1382 answer.ptr = (first += 3);
1383
1384 // The macro casts the literal AFTER applying the negative sign. This is ok:
1385 // -NAN is a double negative NaN, and (float)(-NAN) correctly produces a negative float NaN.
1386 // Same for infinity — (float)(-INFINITY) gives negative float infinity
1387 ffc_set_value(value, vk, minus_sign ? -NAN : NAN);
1388 // Check for possible nan(n-char-seq-opt), C++17 20.19.3.7,
1389 // C11 7.20.1.3.3. At least MSVC produces nan(ind) and nan(snan).
1390 if (first != last && *first == '(') {
1391 for (char *ptr = first + 1; ptr != last; ++ptr) {
1392 if (*ptr == ')') {
1393 answer.ptr = ptr + 1; // valid nan(n-char-seq-opt)
1394 break;
1395 } else if (!(('a' <= *ptr && *ptr <= 'z') ||
1396 ('A' <= *ptr && *ptr <= 'Z') ||
1397 ('0' <= *ptr && *ptr <= '9') || *ptr == '_'))
1398 break; // forbidden char, not nan(n-char-seq-opt)
1399 }
1400 }
1401 return answer;
1402 }
1403 if (ffc_strncasecmp3(first, "infinity", 1)) {
1404 if ((last - first >= 8) &&
1405 ffc_strncasecmp5(first + 3, (char*)&"infinity"[3], 1)) {
1406 answer.ptr = first + 8;
1407 } else {
1408 answer.ptr = first + 3;
1409 }
1410 ffc_set_value(value, vk, minus_sign ? -INFINITY : INFINITY);
1411 return answer;
1412 }
1413 }
1414 answer.outcome = FFC_OUTCOME_INVALID_INPUT;
1415 return answer;
1416}

1418ffc_internal ffc_inline
1419ffc_result ffc_parse_int_string(
1420 char const *p,
1421 char const *pend,
1422 ffc_int_value *value,
1423 ffc_int_kind ik,
1424 ffc_parse_options options,
1425 int const base
1426 ) {
1427 ffc_debug("input '%.*s'... ", (int)(pend - p), p);
1428 ffc_format const fmt = options.format;
1429
1430 if ((uint64_t)(fmt & FFC_FORMAT_FLAG_SKIP_WHITE_SPACE)) {
1431 while ((p != pend) && ffc_is_space(*p)) {
1432 p++;
1433 }
1434 }
1435
1436 if (p == pend || base < 2 || base > 36) {
1437 ffc_result invalid_input_result;
1438 invalid_input_result.ptr = (char*)p;
1439 invalid_input_result.outcome = FFC_OUTCOME_INVALID_INPUT;
1440 return invalid_input_result;
1441 }
1442
1443 ffc_result answer;
1444 char const *const first = p;
1445
1446 bool const negative = (*p == (char)('-'));
1447
1448 if (!ffc_int_kind_is_signed(ik) && negative) {
1449 answer.outcome = FFC_OUTCOME_INVALID_INPUT;
1450 answer.ptr = (char*)first;
1451 return answer;
1452 }
1453 if ((*p == (char)('-')) ||
1454 ((uint64_t)(fmt & FFC_FORMAT_FLAG_ALLOW_LEADING_PLUS) && (*p == (char)('+')))) {
1455 ++p;
1456 }
1457
1458 char const *const start_num = p;
1459
1460 while (p != pend && *p == (char)('0')) {
1461 ++p;
1462 }
1463
1464 bool const has_leading_zeros = p > start_num;
1465
1466 char const *const start_digits = p;
1467
1468 uint64_t i = 0;
1469 if (base == 10) {
1470 ffc_loop_parse_if_eight_digits(&p, pend, &i); // use SIMD if possible
1471 }
1472 while (p != pend) {
1473 uint8_t digit = ffc_char_to_digit(*p);
1474 if (digit >= base) {
1475 break;
1476 }
1477 i = (uint64_t)(base) * i + digit; // might overflow, check this later
1478 p++;
1479 }
1480
1481 size_t digit_count = (size_t)(p - start_digits);
1482
1483 if (digit_count == 0) {
1484 if (has_leading_zeros) {
1485 value->u64 = 0; // Must zero the largest variant!
1486 answer.outcome = FFC_OUTCOME_OK;
1487 answer.ptr = p;
1488 } else {
1489 answer.outcome = FFC_OUTCOME_INVALID_INPUT;
1490 answer.ptr = first;
1491 }
1492 return answer;
1493 }
1494
1495 answer.ptr = p;
1496
1497 // check u64 overflow
1498 size_t max_digits = ffc_max_digits_u64(base);
1499 ffc_debug("digit_count %d, max_digits: %d\n", digit_count, max_digits);
1500 if (digit_count > max_digits) {
1501 answer.outcome = FFC_OUTCOME_OUT_OF_RANGE;
1502 return answer;
1503 }
1504 // this check can be eliminated for all other types, but they will all require
1505 // a max_digits(base) equivalent
1506 if (digit_count == max_digits && i < ffc_min_safe_u64_of_base(base)) {
1507 answer.outcome = FFC_OUTCOME_OUT_OF_RANGE;
1508 return answer;
1509 }
1510
1511 ffc_debug("i is %lld, ik is %s\n", i, (ik == FFC_INT_KIND_U64 ? "u64" :
1512 ik == FFC_INT_KIND_S64 ? "s64" :
1513 ik == FFC_INT_KIND_U32 ? "u32" :
1514 ik == FFC_INT_KIND_S32 ? "s32" : "?"));
1515 // check other types overflow
1516 if (ik != FFC_INT_KIND_U64) {
1517 // Allow 1 greater magnitude when negative
1518 if (i > ffc_int_value_max(ik) + (uint64_t)(negative)) {
1519 answer.outcome = FFC_OUTCOME_OUT_OF_RANGE;
1520 return answer;
1521 }
1522 }
1523
1524 // All signed conversion goes through unsigned arithmetic to avoid UB
1525 if (negative) {
1526 uint64_t neg_i = ~i + 1; // This is the two's complement negation
1527 // write into the signed slot via union
1528 switch (ik) {
1529 case FFC_INT_KIND_S64: value->s64 = (int64_t)neg_i; break; // implementation-defined, but works everywhere
1530 case FFC_INT_KIND_S32: value->s32 = (int32_t)(int64_t)neg_i; break;
1531 // unsigned kinds can't be negative — guarded by the range check above
1532 // case FFC_INT_KIND_S16: value.i16 = (int16_t)(int64_t)neg_i; break;
1533 // case FFC_INT_KIND_S8: value.i8 = (int8_t)(int64_t)neg_i; break;
1534 }
1535 } else {
1536 switch (ik) {
1537 case FFC_INT_KIND_S64: value->s64 = (int64_t)i; break;
1538 case FFC_INT_KIND_S32: value->s32 = (int32_t)(int64_t)i; break;
1539 case FFC_INT_KIND_U64: value->u64 = i; break;
1540 case FFC_INT_KIND_U32: value->u32 = (uint32_t)i; break;
1541 }
1542 }
1543
1544 answer.outcome = FFC_OUTCOME_OK;
1545 return answer;
1546}

1687// try to reserve new_len limbs, filling with limbs
1688// FF_DIVERGE: We remove some extra helpers and simply fill with zeros
1689ffc_internal ffc_inline
1690bool ffc_sv_try_reserve(ffc_sv* sv, size_t new_len) {
1691 if (new_len > SV_LIMB_COUNT) {
1692 return false;

1697 ffc_bigint_limb *last = first + fill_count;
1698 for (ffc_bigint_limb* p = first; p < last; p++) {
1699 *p = 0;
1700 }
1701 sv->len = (uint16_t)new_len;
1702 } else {
1703 sv->len = (uint16_t)new_len;
1704 }
1705 return true;

1737ffc_internal ffc_inline
1738uint64_t ffc_uint64_hi64_2(uint64_t r0, uint64_t r1, bool* truncated) {
1739 FFC_DEBUG_ASSERT(r0 != 0);
1740 int shl = (int)ffc_count_leading_zeroes(r0);
1741 if (shl == 0) {
1742 *truncated = r1 != 0;
1743 return r0;
1744 } else {
1745 int shr = 64 - shl;
1746 *truncated = (r1 << shl) != 0;
1747 return (r0 << shl) | (r1 >> shr);
1748 }
1749}

1771// add two small integers, checking for overflow.
1772// we want an efficient operation. for msvc, where
1773// we don't have built-in intrinsics, this is still
1774// pretty fast.
1775ffc_internal ffc_inline
1776ffc_bigint_limb ffc_bigint_scalar_add(ffc_bigint_limb x, ffc_bigint_limb y, bool* overflow) {
1777 ffc_bigint_limb z;
1778// gcc and clang
1779#if defined(__has_builtin)
1780#if __has_builtin(__builtin_add_overflow)
1781 *overflow = __builtin_add_overflow(x, y, &z);
1782 return z;
1783#endif
1784#endif
1785
1786 // generic, this still optimizes correctly on MSVC.
1787 z = x + y;
1788 *overflow = z < x;
1789 return z;
1790}

1792// multiply two small integers, getting both the high and low bits.
1793ffc_inline ffc_internal
1794ffc_bigint_limb ffc_bigint_scalar_mul(ffc_bigint_limb x, ffc_bigint_limb y, ffc_bigint_limb* carry) {
1795#ifdef FFC_64BIT_LIMB
1796#if defined(__SIZEOF_INT128__)
1797 // GCC and clang both define it as an extension.
1798 __uint128_t z = (__uint128_t)(x) * (__uint128_t)(y) + (__uint128_t)(*carry);
1799 *carry = (ffc_bigint_limb)(z >> FFC_LIMB_BITS);
1800 return (ffc_bigint_limb)(z);
1801#else
1802 // fallback, no native 128-bit integer multiplication with carry.
1803 // on msvc, this optimizes identically, somehow.
1804 ffc_u128 z = ffc_full_multiplication(x, y);
1805 bool overflow;
1806 z.low = ffc_bigint_scalar_add(z.low, *carry, &overflow);
1807 z.high += (uint64_t)(overflow); // cannot overflow
1808 *carry = z.high;
1809 return z.low;
1810#endif
1811#else
1812 uint64_t z = (uint64_t)(x) * (uint64_t)(y) + (uint64_t)(*carry);
1813 *carry = (ffc_bigint_limb)(z >> FFC_LIMB_BITS);
1814 return (ffc_bigint_limb)(z);
1815#endif
1816}

1818// add scalar value to bigint starting from offset.
1819// used in grade school multiplication
1820ffc_internal ffc_inline
1821bool ffc_bigint_small_add_from(ffc_sv* sv, ffc_bigint_limb y, size_t start) {
1822 size_t index = start;
1823 ffc_bigint_limb carry = y;
1824 bool overflow;
1825 while (carry != 0 && index < sv->len) {
1826 sv->data[index] = ffc_bigint_scalar_add(sv->data[index], carry, &overflow);
1827 carry = (ffc_bigint_limb)(overflow);
1828 index += 1;
1829 }
1830 if (carry != 0) {
1831 FFC_TRY(ffc_sv_try_push(sv, carry));
1832 }
1833 return true;
1834}

1855// add bigint to bigint starting from index.
1856// used in grade school multiplication
1857ffc_internal
1858bool ffc_bigint_large_add_from(ffc_sv* x, ffc_bigint_limb_span y, size_t start) {
1859 // the effective x buffer is from `xstart..x.len()`, so exit early
1860 // if we can't get that current range.
1861
1862 // FFC_DIVERGE: We are calling our try_reserve instead of the o.g. try_resize
1863 if (x->len < start || y.len > x->len - start) {
1864 FFC_TRY(ffc_sv_try_reserve(x, y.len + start));
1865 }
1866
1867 bool carry = false;
1868 for (size_t index = 0; index < y.len; index++) {
1869 ffc_bigint_limb xi = x->data[index + start];
1870 ffc_bigint_limb yi = ffc_span_index(y, index);
1871 bool c1 = false;
1872 bool c2 = false;
1873 xi = ffc_bigint_scalar_add(xi, yi, &c1);
1874 if (carry) {
1875 xi = ffc_bigint_scalar_add(xi, 1, &c2);
1876 }
1877 x->data[index + start] = xi;
1878 carry = c1 | c2;
1879 }
1880
1881 // handle overflow
1882 if (carry) {
1883 FFC_TRY(ffc_bigint_small_add_from(x, 1, y.len + start));
1884 }
1885 return true;
1886}

1894// grade-school multiplication algorithm
1895ffc_internal
1896bool ffc_bigint_long_mul(ffc_sv* x, ffc_bigint_limb_span y) {
1897 ffc_bigint_limb_span xs;
1898 xs.ptr = x->data;
1899 xs.len = x->len;
1900
1901 // full copy of x into z
1902 ffc_sv z = ffc_sv_create(xs);
1903
1904 ffc_bigint_limb_span zs;
1905 zs.ptr = z.data;
1906 zs.len = z.len;
1907
1908 if (y.len != 0) {
1909 ffc_bigint_limb y0 = ffc_span_index(y, 0);
1910 FFC_TRY(ffc_bigint_small_mul(x, y0));
1911 for (size_t index = 1; index < y.len; index++) {
1912
1913 ffc_bigint_limb yi = ffc_span_index(y, index);
1914 ffc_sv zi; // re-use the same buffer throughout
1915
1916 if (yi != 0) {
1917 zi.len = 0;
1918 FFC_TRY(ffc_sv_try_extend(&zi, zs));
1919 FFC_TRY(ffc_bigint_small_mul(&zi, yi));
1920 ffc_bigint_limb_span zis;
1921 zis.ptr = zi.data;
1922 zis.len = zi.len;
1923 FFC_TRY(ffc_bigint_large_add_from(x, zis, index));
1924 }
1925 }
1926 }
1927
1928 ffc_sv_normalize(x);
1929 return true;
1930}

2057// compare two big integers, returning the large value.
2058// assumes both are normalized. if the return value is
2059// negative, other is larger, if the return value is
2060// positive, this is larger, otherwise they are equal.
2061// the limbs are stored in little-endian order, so we
2062// must compare the limbs in ever order.
2063ffc_internal ffc_inline
2064int ffc_bigint_compare(ffc_bigint me, ffc_bigint const *other) {
2065 if (me.vec.len > other->vec.len) {
2066 return 1;
2067 } else if (me.vec.len < other->vec.len) {
2068 return -1;
2069 } else {
2070 for (size_t index = me.vec.len; index > 0; index--) {
2071 ffc_bigint_limb xi = ffc_sv_index(me.vec, index - 1);
2072 ffc_bigint_limb yi = ffc_sv_index(other->vec, index - 1);
2073 if (xi > yi) {
2074 return 1;
2075 } else if (xi < yi) {
2076 return -1;
2077 }
2078 }
2079 return 0;
2080 }
2081}

2083// shift left each limb n bits, carrying over to the new limb
2084// returns true if we were able to shift all the digits.
2085ffc_internal ffc_inline
2086bool ffc_bigint_shl_bits(ffc_bigint* me, size_t n) {
2087 // Internally, for each item, we shift left by n, and add the previous
2088 // right shifted limb-bits.
2089 // For example, we transform (for u8) shifted left 2, to:
2090 // b10100100 b01000010
2091 // b10 b10010001 b00001000
2092 FFC_DEBUG_ASSERT(n != 0);
2093 FFC_DEBUG_ASSERT(n < sizeof(ffc_bigint_limb) * 8);
2094
2095 size_t shl = n;
2096 size_t shr = FFC_LIMB_BITS - shl;
2097 ffc_bigint_limb prev = 0;
2098 for (size_t index = 0; index < me->vec.len; index++) {
2099 ffc_bigint_limb xi = ffc_sv_index(me->vec, index);
2100 me->vec.data[index] = (xi << shl) | (prev >> shr);
2101 prev = xi;
2102 }
2103
2104 ffc_bigint_limb carry = prev >> shr;
2105 if (carry != 0) {
2106 return ffc_sv_try_push(&me->vec, carry);
2107 }
2108 return true;
2109}

2111// move the limbs left by `n` limbs.
2112ffc_internal ffc_inline
2113bool ffc_bigint_shl_limbs(ffc_bigint* me, size_t n) {
2114 FFC_DEBUG_ASSERT(n != 0);
2115 if (n + me->vec.len > SV_LIMB_COUNT) {
2116 return false;
2117 } else if (me->vec.len != 0) {
2118 // move limbs
2119 ffc_bigint_limb *dst = me->vec.data + n;
2120 ffc_bigint_limb const *src = me->vec.data;
2121 // std::copy_backward(src, src + vec.len(), dst + vec.len());
2122 // memmove to handle the overlap
2123 memmove(dst, src, me->vec.len * sizeof(ffc_bigint_limb));
2124
2125 // fill in empty limbs
2126 ffc_bigint_limb *first = me->vec.data;
2127 // ffc_bigint_limb *last = first + n;
2128 // ::std::fill(first, last, 0);
2129 memset(first, 0, n * sizeof(ffc_bigint_limb));
2130 me->vec.len += n;
2131 return true;
2132 } else {
2133 return true;
2134 }
2135}

2273// this converts a native floating-point number to an extended-precision float.
2274ffc_internal ffc_inline ffc_adjusted_mantissa
2275ffc_to_extended(ffc_value value, ffc_value_kind vk) {
2276 uint64_t const exponent_mask = ffc_const(vk, EXPONENT_MASK);
2277 uint64_t const mantissa_mask = ffc_const(vk, MANTISSA_MASK);
2278 uint64_t const hidden_bit_mask = ffc_const(vk, HIDDEN_BIT_MASK);
2279
2280 // Finish converting this one
2281 ffc_adjusted_mantissa am;
2282 int32_t bias = ffc_const(vk, MANTISSA_EXPLICIT_BITS) - ffc_const(vk, MINIMUM_EXPONENT);
2283 //ffc_int_value bits = ffc_get_value_bits(value, vk);
2284
2285 switch (vk) {
2286 case FFC_VALUE_KIND_DOUBLE: {
2287 uint64_t bits = ffc_get_double_bits(value.d);
2288 if ((bits & exponent_mask) == 0) {
2289 // denormal
2290 am.power2 = 1 - bias;
2291 am.mantissa = bits & mantissa_mask;
2292 } else {
2293 // normal
2294 am.power2 = (int32_t)((bits & exponent_mask) >> ffc_const(vk, MANTISSA_EXPLICIT_BITS));
2295 am.power2 -= bias;
2296 am.mantissa = (bits & mantissa_mask) | hidden_bit_mask;
2297 }
2298 break;
2299 }
2300 case FFC_VALUE_KIND_FLOAT: {
2301 uint32_t bits = ffc_get_float_bits(value.f);
2302 if ((bits & exponent_mask) == 0) {
2303 // denormal
2304 am.power2 = 1 - bias;
2305 am.mantissa = bits & mantissa_mask;
2306 } else {
2307 // normal
2308 am.power2 = (int32_t)((bits & exponent_mask) >> ffc_const(vk, MANTISSA_EXPLICIT_BITS));
2309 am.power2 -= bias;
2310 am.mantissa = (bits & mantissa_mask) | hidden_bit_mask;
2311 }
2312 break;
2313 }
2314 }

2316 return am;
2317}

2341// round nearest tie even, resolving ties using the truncated flag
2342ffc_internal ffc_inline
2343void ffc_round_nearest_tie_even_truncated(ffc_adjusted_mantissa *am,
2344 int32_t shift, bool truncated) {
2345 uint64_t const mask = (shift == 64) ? UINT64_MAX : ((uint64_t)(1) << shift) - 1;
2346 uint64_t const halfway = (shift == 0) ? 0 : (uint64_t)(1) << (shift - 1);
2347 uint64_t truncated_bits = am->mantissa & mask;
2348 bool is_above = truncated_bits > halfway;
2349 bool is_halfway = truncated_bits == halfway;
2350
2351 // shift digits into position
2352 if (shift == 64) {
2353 am->mantissa = 0;
2354 } else {
2355 am->mantissa >>= shift;
2356 }
2357 am->power2 += shift;
2358
2359 bool is_odd = (am->mantissa & 1) == 1;
2360 am->mantissa += (uint64_t)(is_above || (is_halfway && truncated) ||
2361 (is_odd && is_halfway));

2380// Finalize rounding for a double after the shift callback has been applied.
2381// This is the common tail of ffc_round_double factored out to avoid
2382// duplicating the carry/infinite checks at every call site.
2383ffc_internal ffc_inline
2384void ffc_round_finish(ffc_adjusted_mantissa *am, ffc_value_kind vk) {
2385 // check for carry
2386 if (am->mantissa >= ((uint64_t)(2) << ffc_const(vk, MANTISSA_EXPLICIT_BITS))) {
2387 am->mantissa = ((uint64_t)(1) << ffc_const(vk, MANTISSA_EXPLICIT_BITS));
2388 am->power2++;
2389 }
2390
2391 // check for infinite: we could have carried to an infinite power
2392 am->mantissa &= ~((uint64_t)(1) << ffc_const(vk, MANTISSA_EXPLICIT_BITS));
2393 if (am->power2 >= ffc_const(vk, INFINITE_POWER)) {
2394 am->power2 = ffc_const(vk, INFINITE_POWER);
2395 am->mantissa = 0;
2396 }
2397}

2566// parse the significant digits into a big integer
2567ffc_internal ffc_inline
2568void ffc_parse_mantissa(ffc_bigint *result, ffc_parsed num,
2569 size_t max_digits, size_t *const digits) {
2570 // try to minimize the number of big integer and scalar multiplication.
2571 // therefore, try to parse 8 digits at a time, and multiply by the largest
2572 // scalar value (9 or 19 digits) for each step.
2573 size_t counter = 0;
2574 *digits = 0;
2575 ffc_bigint_limb value = 0;
2576#ifdef FFC_64BIT_LIMB
2577 size_t step = 19;
2578#else
2579 size_t step = 9;
2580#endif
2581
2582 // process all integer digits.
2583 char *p = num.int_part_start;
2584 char *pend = p + num.int_part_len;
2585 ffc_skip_zeros(&p, pend, 1);
2586 // process all digits, in increments of step per loop
2587 while (p != pend) {
2588 while ((pend - p >= 8) && (step - counter >= 8) &&
2589 (max_digits - *digits >= 8)) {
2590 ffc_parse_eight_digits(&p, &value, &counter, digits);
2591 }
2592 while (counter < step && p != pend && *digits < max_digits) {
2593 ffc_parse_one_digit(&p, &value, &counter, digits);
2594 }
2595 if (*digits == max_digits) {
2596 // add the temporary value, then check if we've truncated any digits
2597 ffc_add_native(result, (ffc_bigint_limb)(ffc_powers_of_ten_uint64[counter]), value);
2598 bool truncated = ffc_is_truncated(p, pend, 1);
2599 if (num.fraction_part_start != NULL) {
2600 truncated |= ffc_is_truncated(num.fraction_part_start, num.fraction_part_start + num.fraction_part_len, 1);
2601 }
2602 if (truncated) {
2603 ffc_round_up_bigint(result, digits);
2604 }
2605 return;
2606 } else {
2607 ffc_add_native(result, (ffc_bigint_limb)(ffc_powers_of_ten_uint64[counter]), value);
2608 counter = 0;
2609 value = 0;
2610 }
2611 }
2612
2613 // add our fraction digits, if they're available.
2614 if (num.fraction_part_start != NULL) {
2615 p = num.fraction_part_start;
2616 pend = p + num.fraction_part_len;
2617 if (*digits == 0) {
2618 ffc_skip_zeros(&p, pend, 1);
2619 }
2620 // process all digits, in increments of step per loop
2621 while (p != pend) {
2622 while ((pend - p >= 8) && (step - counter >= 8) &&
2623 (max_digits - *digits >= 8)) {
2624 ffc_parse_eight_digits(&p, &value, &counter, digits);
2625 }
2626 while (counter < step && p != pend && *digits < max_digits) {
2627 ffc_parse_one_digit(&p, &value, &counter, digits);
2628 }
2629 if (*digits == max_digits) {
2630 // add the temporary value, then check if we've truncated any digits
2631 ffc_add_native(result, (ffc_bigint_limb)(ffc_powers_of_ten_uint64[counter]), value);
2632 bool truncated = ffc_is_truncated(p, pend, 1);
2633 if (truncated) {
2634 ffc_round_up_bigint(result, digits);
2635 }
2636 return;
2637 } else {
2638 ffc_add_native(result, (ffc_bigint_limb)(ffc_powers_of_ten_uint64[counter]), value);
2639 counter = 0;
2640 value = 0;
2641 }
2642 }
2643 }
2644
2645 if (counter != 0) {
2646 ffc_add_native(result, (ffc_bigint_limb)(ffc_powers_of_ten_uint64[counter]), value);
2647 }
2648}

2667// the scaling here is quite simple: we have, for the real digits `m * 10^e`,
2668// and for the theoretical digits `n * 2^f`. Since `e` is always negative,
2669// to scale them identically, we do `n * 2^f * 5^-f`, so we now have `m * 2^e`.
2670// we then need to scale by `2^(f- e)`, and then the two significant digits
2671// are of the same magnitude.
2672ffc_internal ffc_inline
2673ffc_adjusted_mantissa ffc_negative_digit_comp(
2674 ffc_bigint *bigmant, ffc_adjusted_mantissa am, int32_t exponent, ffc_value_kind vk) {
2675 ffc_bigint *real_digits = bigmant;
2676 int32_t real_exp = exponent;
2677
2678 // get the value of `b`, rounded down, and get a bigint representation of b+h
2679 ffc_adjusted_mantissa am_b = am;
2680 ffc_round_down(&am_b, vk);
2681 ffc_value b;
2682 ffc_am_to_float(false, am_b, &b, FFC_VALUE_KIND_DOUBLE);
2683
2684 ffc_adjusted_mantissa theor = ffc_to_extended_halfway(b, vk);
2685 ffc_bigint theor_digits = ffc_bigint_make(theor.mantissa);
2686 int32_t theor_exp = theor.power2;
2687
2688 // scale real digits and theor digits to be same power.
2689 int32_t pow2_exp = theor_exp - real_exp;
2690 uint32_t pow5_exp = (uint32_t)(-real_exp);
2691 if (pow5_exp != 0) {
2692 FFC_ASSERT(ffc_bigint_pow5(&theor_digits, pow5_exp));
2693 }
2694 if (pow2_exp > 0) {
2695 FFC_ASSERT(ffc_bigint_pow2(&theor_digits, (uint32_t)(pow2_exp)));
2696 } else if (pow2_exp < 0) {
2697 FFC_ASSERT(ffc_bigint_pow2(real_digits,(uint32_t)(-pow2_exp)));
2698 }
2699
2700 // compare digits, and use it to direct rounding
2701 int ord = ffc_bigint_compare(*real_digits, &theor_digits);
2702 ffc_adjusted_mantissa answer = am;
2703 ffc_round_double_tie_even_cmp(&answer, ord, vk);
2704
2705 return answer;
2706}

2708// parse the significant digits as a big integer to unambiguously round
2709// the significant digits. here, we are trying to determine how to round
2710// an extended float representation close to `b+h`, halfway between `b`
2711// (the float rounded-down) and `b+u`, the next positive float. this
2712// algorithm is always correct, and uses one of two approaches. when
2713// the exponent is positive relative to the significant digits (such as
2714// 1234), we create a big-integer representation, get the high 64-bits,
2715// determine if any lower bits are truncated, and use that to direct
2716// rounding. in case of a negative exponent relative to the significant
2717// digits (such as 1.2345), we create a theoretical representation of
2718// `b` as a big-integer type, scaled to the same binary exponent as
2719// the actual digits. we then compare the big integer representations
2720// of both, and use that to direct rounding.
2721ffc_internal
2722ffc_adjusted_mantissa ffc_digit_comp(ffc_parsed num, ffc_adjusted_mantissa am, ffc_value_kind vk) {
2723 ffc_debug("digit_comp\n");
2724 // remove the invalid exponent bias
2725 am.power2 -= FFC_INVALID_AM_BIAS;
2726
2727 int32_t sci_exp =
2728 ffc_scientific_exponent(num.mantissa, (int32_t)(num.exponent));
2729 size_t max_digits = ffc_const(vk, MAX_DIGITS);
2730 size_t digits = 0;
2731 ffc_bigint bigmant = ffc_bigint_empty();
2732 ffc_parse_mantissa(&bigmant, num, max_digits, &digits);
2733 // can't underflow, since digits is at most max_digits.
2734 int32_t exponent = sci_exp + 1 - (int32_t)(digits);
2735 if (exponent >= 0) {
2736 return ffc_positive_digit_comp(&bigmant, exponent, vk);
2737 } else {
2738 return ffc_negative_digit_comp(&bigmant, am, exponent, vk);
2739 }
2740}

2746ffc_inline ffc_internal
2747ffc_u128 ffc_compute_product_approximation(int64_t q, uint64_t w, ffc_value_kind vk) {
2748 // The required precision is mantissa_explicit_bits + 3 because
2749 // 1. We need the implicit bit
2750 // 2. We need an extra bit for rounding purposes
2751 // 3. We might lose a bit due to the "upperbit" routine (result too small,
2752 // requiring a shift)
2753 uint64_t bit_precision = ffc_const(vk, MANTISSA_EXPLICIT_BITS) + 3;
2754 uint64_t precision_mask = ((uint64_t)(0xFFFFFFFFFFFFFFFF) >> bit_precision);
2755
2756 // Use FFC_DOUBLE_SMALLEST_POWER_OF_10 (-342) regardless of value kind
2757 // to index into the shared 128 bit table
2758 int const index = 2 * (int)(q - FFC_DOUBLE_SMALLEST_POWER_OF_10);
2759
2760 // For small values of q, e.g., q in [0,27], the answer is always exact
2761 // because ffc_mul_u64(w, powers_of_five[index]) gives the exact answer.
2762 ffc_u128 firstproduct = ffc_mul_u64(w, FFC_POWERS_OF_FIVE[index]);
2763
2764 if ((firstproduct.high & precision_mask) == precision_mask) {
2765 // could further guard with (lower + w < lower)
2766 // regarding the second product, we only need secondproduct.hi, but our
2767 // expectation is that the compiler will optimize this extra work away if
2768 // needed.
2769 ffc_u128 secondproduct = ffc_mul_u64(w, FFC_POWERS_OF_FIVE[index + 1]);
2770
2771 firstproduct.low += secondproduct.high;
2772 if (secondproduct.high > firstproduct.low) {
2773 firstproduct.high++;
2774 }
2775 }
2776 return firstproduct;
2777}

2814 if (q > ffc_const(vk, LARGEST_POWER_OF_10)) {
2815 // we want to get infinity:
2816 answer.power2 = ffc_const(vk, INFINITE_POWER);
2817 answer.mantissa = 0;
2818 return answer;
2819 }
2820 // At this point in time q is in [powers::smallest_power_of_five,
2821 // powers::largest_power_of_five].
2822
2823 // We want the most significant bit of i to be 1. Shift if needed.
2824 int32_t lz = (int32_t)ffc_count_leading_zeroes(w);
2825 w <<= lz;
2826
2827 ffc_u128 product = ffc_compute_product_approximation(q, w, vk);
2828
2829 // The computed 'product' is always sufficient.
2830 // Mathematical proof:
2831 // Noble Mushtak and Daniel Lemire, Fast Number Parsing Without Fallback (to
2832 // appear) See script/mushtak_lemire.py
2833
2834 // The "compute_product_approximation" function can be slightly slower than a
2835 // branchless approach: value128 product = compute_product(q, w); but in
2836 // practice, we can win big with the compute_product_approximation if its
2837 // additional branch is easily predicted. Which is data specific.
2838 int upperbit = (int)(product.high >> 63);
2839 int shift = upperbit + 64 - ffc_const(vk, MANTISSA_EXPLICIT_BITS) - 3;
2840
2841 answer.mantissa = product.high >> shift;
2842
2843 // compute a biased-up power of 2
2844 answer.power2 = (int32_t)(ffc_b10_to_b2((int32_t)(q)) + upperbit - lz - ffc_const(vk, MINIMUM_EXPONENT));
2845
2846 if (answer.power2 <= 0) { // subnormal path
2847
2848 if (-answer.power2 + 1 >= 64) {
2849 // if we have more than 64 bits below the minimum exponent, you
2850 // have a zero for sure.
2851 answer.power2 = 0;
2852 answer.mantissa = 0;
2853 // result should be zero
2854 return answer;
2855 }
2856 // shift is safe because -answer.power2 + 1 < 64
2857 answer.mantissa >>= -answer.power2 + 1;
2858
2859 // Thankfully, we can't have both "round-to-even" and subnormals because
2860 // "round-to-even" only occurs for powers close to 0 in the 32-bit and
2861 // and 64-bit case (with no more than 19 digits), so we round up.
2862 answer.mantissa += (answer.mantissa & 1); // round up
2863 answer.mantissa >>= 1;
2864
2865 // weird scenario:
2866 // Suppose we start with 2.2250738585072013e-308, we end up
2867 // with 0x3fffffffffffff x 2^-1023-53 which is technically subnormal
2868 // whereas 0x40000000000000 x 2^-1023-53 is normal. Now, we need to round
2869 // up 0x3fffffffffffff x 2^-1023-53 and once we do, we are no longer
2870 // subnormal, but we can only know this after rounding.
2871 // So we only declare a subnormal if we are smaller than the threshold.
2872 answer.power2 =
2873 (answer.mantissa < ((uint64_t)(1) << ffc_const(vk, MANTISSA_EXPLICIT_BITS))) ? 0 : 1;
2874 return answer;
2875 } // subnormal
2876
2877 // usually, we round *up*, but if we fall right in between and we have an
2878 // even basis, we need to round down
2879 // We are only concerned with the cases where 5**q fits in single 64-bit word.
2880
2881 // 'extremely sparse low bits in our product' and
2882 // 'q is within the round to even range' and
2883 // 'mantissa lowest 2 bits are exactly 01'
2884 if ((product.low <= 1) && (q >= ffc_const(vk, MIN_EXPONENT_ROUND_TO_EVEN)) &&
2885 (q <= ffc_const(vk, MAX_EXPONENT_ROUND_TO_EVEN)) &&
2886 ((answer.mantissa & 3) == 1)) { // we may fall between two floats!
2887 //
2888 // To be in-between two floats we need that in doing
2889 // answer.mantissa = product.high >> (upperbit + 64 -
2890 // binary::mantissa_explicit_bits() - 3);
2891 // ... we dropped out only zeroes. But if this happened, then we can go
2892 // back!!!
2893
2894 // mask off last bit
2895 if ((answer.mantissa << shift) == product.high) {
2896 answer.mantissa &= ~(uint64_t)(1);
2897 }
2898 }
2899
2900 answer.mantissa += (answer.mantissa & 1); // round up
2901 answer.mantissa >>= 1;
2902 if (answer.mantissa >= ((uint64_t)(2) << ffc_const(vk, MANTISSA_EXPLICIT_BITS))) {
2903 answer.mantissa = ((uint64_t)(1) << ffc_const(vk, MANTISSA_EXPLICIT_BITS));
2904 answer.power2++; // undo previous addition
2905 }
2906
2907 // normalize to pos INF?
2908 answer.mantissa &= ~((uint64_t)(1) << ffc_const(vk, MANTISSA_EXPLICIT_BITS));
2909 if (answer.power2 >= ffc_const(vk, INFINITE_POWER)) { // infinity
2910 answer.power2 = ffc_const(vk, INFINITE_POWER);
2911 answer.mantissa = 0;
2912 }
2913 return answer;
2914}

2919ffc_adjusted_mantissa ffc_compute_error_scaled(int64_t q, uint64_t w, int lz, ffc_value_kind vk) {

2921 ffc_adjusted_mantissa answer;

2924 answer.power2 = (int32_t)(ffc_b10_to_b2((int32_t)(q)) + bias - hilz - lz - 62 +
2925 FFC_INVALID_AM_BIAS);

2943ffc_internal ffc_inline
2944bool ffc_clinger_fast_path_impl(uint64_t mantissa, int64_t exponent, bool is_negative,
2945 ffc_value *value, ffc_value_kind value_kind) {
2946 bool is_double = value_kind == FFC_VALUE_KIND_DOUBLE;
2947 // The implementation of the Clinger's fast path is convoluted because
2948 // we want round-to-nearest in all cases, irrespective of the rounding mode
2949 // selected on the thread.
2950 // We proceed optimistically, assuming that detail::rounds_to_nearest()
2951 // returns true.
2952 if (ffc_const(value_kind, MIN_EXPONENT_FAST_PATH) <= exponent &&
2953 exponent <= ffc_const(value_kind, MAX_EXPONENT_FAST_PATH)) {
2954 // Unfortunately, the conventional Clinger's fast path is only possible
2955 // when the system rounds to the nearest float.
2956 //
2957 // We expect the next branch to almost always be selected.
2958 // We could check it first (before the previous branch), but
2959 // there might be performance advantages at having the check
2960 // be last.
2961 if (ffc_rounds_to_nearest()) {
2962 // We have that fegetround() == FE_TONEAREST.
2963 // Next is Clinger's fast path.
2964 if (mantissa <= ffc_const(value_kind, MAX_MANTISSA_FAST_PATH)) {
2965 ffc_set_value(value, value_kind, mantissa);
2966
2967 if (exponent < 0) {
2968 if (is_double) {
2969 value->d = value->d / FFC_DOUBLE_POWERS_OF_TEN[-exponent];
2970 } else {
2971 value->f = value->f / FFC_FLOAT_POWERS_OF_TEN[-exponent];
2972 };
2973 } else {
2974 if (is_double) {
2975 value->d = value->d * FFC_DOUBLE_POWERS_OF_TEN[exponent];
2976 } else {
2977 value->f = value->f * FFC_FLOAT_POWERS_OF_TEN[exponent];
2978 };
2979 }
2980 if (is_negative) {
2981 ffc_set_value(value, value_kind, -ffc_read_value(value, value_kind));
2982 }
2983 return true;
2984 }
2985 } else {
2986 // We do not have that fegetround() == FE_TONEAREST.
2987 // Next is a modified Clinger's fast path, inspired by Jakub Jelínek's
2988 // proposal
2989 if (exponent >= 0 &&
2990 mantissa <= ffc_const(value_kind, MAX_MANTISSA)[exponent]) {
2991#if defined(__clang__) || defined(FFC_32BIT)
2992 // Clang may map 0 to -0.0 when fegetround() == FE_DOWNWARD
2993 if (mantissa == 0) {
2994 ffc_set_value(value, value_kind, is_negative ? -0. : 0.);
2995 return true;
2996 }
2997#endif
2998 if (is_double) {
2999 value->d = (double)mantissa * FFC_DOUBLE_POWERS_OF_TEN[exponent];
3000 } else {
3001 value->f = (float)mantissa * FFC_FLOAT_POWERS_OF_TEN[exponent];
3002 }
3003 if (is_negative) {
3004 ffc_set_value(value, value_kind, -ffc_read_value(value, value_kind));
3005 }
3006 return true;
3007 }
3008 }
3009 }
3010 return false;
3011}

3014ffc_result ffc_from_chars_advanced(ffc_parsed const pns, ffc_value* value, ffc_value_kind vk) {
3015 ffc_result answer;
3016
3017 answer.outcome = FFC_OUTCOME_OK; // be optimistic :')
3018 answer.ptr = (char*)pns.lastmatch;
3019
3020 if (!pns.too_many_digits &&
3021 ffc_clinger_fast_path_impl(pns.mantissa, pns.exponent, pns.negative, value, vk)) {
3022 ffc_debug("fast path hit");
3023 return answer;
3024 }
3025
3026 ffc_adjusted_mantissa am = ffc_compute_float(pns.exponent, pns.mantissa, vk);
3027 ffc_debug("am.mantissa: %llu\n", am.mantissa);
3028 ffc_debug("am.power2: %d\n", am.power2);
3029 if (pns.too_many_digits && am.power2 >= 0) {
3030 ffc_adjusted_mantissa am_plus_one = ffc_compute_float(pns.exponent, pns.mantissa + 1, vk);
3031 bool equal = am.mantissa == am_plus_one.mantissa && am.power2 == am_plus_one.power2;
3032 if (!equal) {
3033 am = ffc_compute_error(pns.exponent, pns.mantissa, vk);
3034 }
3035 }
3036 // If we called ffc_compute_float(pns.exponent, pns.mantissa)
3037 // and we have an invalid power (am.power2 < 0), then we need to go the long
3038 // way around again. This is very uncommon.
3039 if (am.power2 < 0) {
3040 am = ffc_digit_comp(pns, am, vk);
3041 }
3042 ffc_debug("am post mantissa: %llu\n", am.mantissa);
3043 ffc_debug("am post power2: %d\n", am.power2);
3044 ffc_am_to_float(pns.negative, am, value, vk);
3045
3046 // Test for over/underflow.
3047 if ((pns.mantissa != 0 && am.mantissa == 0 && am.power2 == 0) ||
3048 am.power2 == ffc_const(vk, INFINITE_POWER)) {
3049 answer.outcome = FFC_OUTCOME_OUT_OF_RANGE;
3050 }
3051 return answer;
3052}
